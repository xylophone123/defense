\begin{frame}{Factorization Algebras and Coinvariants}
\begin{itemize}
    \item Vertex algebras model the algebraic structure of conformal field theory in 1 complex dimension.
    \item Factorization algebras (Costello-Gwilliam) were defined to model observables and symmetries of classical and quantum field theory.
    \item It has already been shown (Costello-Gwilliam) that holomoprhic prefactorization algebras can produce vertex algebras.
    \item Vertex algebras have modules and a more interesting related construction of conformal blocks
    \item Can factorization algebras produce modules and recover the space of confomal blocks?
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Prefactorization algebras}
    \begin{block}{Definition (Costello--Gwilliam)}
        For $M$ a topological space a prefactorization algebra on $M$ is an assignment $\mathcal{F}$ taking open sets $U$ of $M$ to objects of a category (vector spaces, cochain complexes), $\mathcal{F}(U)$ with
        \begin{itemize}
            \item A linear map $m^{U}_V :\mathcal{F}(U) \to \mathcal{F}(V)$ for $U\subseteq V$.
            \item For $\{U_i\}$ pairwise disjoint and all contained in $V$, a linear map $m_V^{U_1,\dots,U_n} : \mathcal{F}(U_1)\otimes \dotsb \otimes \mathcal{F}(U_n) \to \mathcal{F}(V)$
            \item The maps need to satisfy the following for $U_{i,1} \sqcup \dotsb \sqcup U_{i,n_i}\subseteq V_i$

                  \begin{tikzcd}
                      \otimes_{i=1}^k \otimes_{j=1}^{n_{i}} \mathcal{F}(U_{i,j}) \arrow[rr] \arrow[rd] &  & \otimes_{i=1}^k \mathcal{F}(V_i) \arrow[dl]\\
                      & \mathcal{F}(W) &
                  \end{tikzcd}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Vertex Algebras}
    \begin{block}{Definition}(Frenkel Ben-Zvi)

        Vertex algebras consist of the following data
        \begin{itemize}
            \item $V$ a vector space
            \item a vacuum vector $|0\rangle \in V$
            \item a (translation) linear operator $T:V\to V$
            \item a linear operation
                  \[
                      Y(-,z):V \to \mbox{End}(V)[[z,z^{-1}]]
                  \]
                  where for each $A,B \in V$,  $Y(A,z)B\in V((z))$
        \end{itemize}
        satisfying certain axioms...
    \end{block}
\end{frame}

\begin{frame}{Vertex Algebras}
    \begin{block}{Definition continued}
        \begin{itemize}
            \item The vacuum axiom $Y(\vac,z) = \mbox{Id}_V$ and for any $A\in V$
                  \[
                      Y(A,z)\vac \in V[[z]], \quad Y(A,z)\vac|_{z=0}=A.
                  \]
            \item The translation axiom, for any $A\in V$,
                  \[
                      [T,Y(A,z)]=\partial_z Y(A,z)
                  \]
                  and $T\vac =0$.
            \item All $Y(A,z)$ are mutually local i.e. for large enough $N$
                  \[
                      [Y(A,z),Y(B,w)](z-w)^N =0
                  \]
            %\item Vertex algebras can be $\Z$-graded
            % \item A vertex algebra is $\Z$ graded if $V$ is $\Z$ graded, $\vac$ is degree 0, $T$ has degree 1, and for $A\in V_m$ and $Y(A,z) = \sum_{\Z} A_{n} z^{-n-1}$ $A_n$ has degree $-n+m-1$
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Affine Kac-Moody Algebra}
    For $\mathfrak{g}$ a finite dimensional simple Lie algebra over $\C$ with canonically scaled killing form
    \[
        (\cdot, \cdot) = \frac{1}{{2h^{\vee}}}(\cdot,\cdot)_{K}
    \]
    where $h^{\vee}$ is the dual Coxeter number of $\mathfrak{g}$. We define $\hat{\mathfrak{g}}$ as the central extension
    \[
        0 \to \C K \to \hat{\mathfrak{g}} \to \mathfrak{g}((t)) \to 0
    \]
    defined by
    \[
        [X\otimes f(t),Y \otimes g(t)]=[X,Y]\otimes f(t)g(t) -(\mbox{Res}_{t=0} f dg(X,Y) K).
    \]
\end{frame}

\begin{frame}{Vacuum Module}
    $\mathfrak{g}_{+}:=\mathfrak{g}[[t]] \oplus \C K$ is a Lie subalgebra of $\hat{\mathfrak{g}}$. For $k\in \C$ we can define $\C_k$ as the representation where $\mathfrak{g}[[t]]$ acts by 0 and $K$ acts by $k$. The \textbf{vacuum module} is the induced $\hat{\mathfrak{g}}$-module of $\C_k$

    \[
        V_{k}(\mathfrak{g}) = \mbox{Ind}_{\mathfrak{g}[[t]]\oplus \C K}^{\hat{\mathfrak{g}}} \C_k = U(\hat{\mathfrak{g}})\otimes_{U(\mathfrak{g}_{+})}\C_k.
    \]
    As a vector space $V_k(\mathfrak{g})\cong U(t^{-1}\mathfrak{g}[t^{-1}])$. For $(J^a)_{a=1}^n$ a basis of $\mathfrak{g}$ and defining
    \[
        J^a_n:= J^a \otimes t^n
    \]
    we can find a PBW basis of $V_k(\mathfrak{g})$ given by letting $v_k = 1 \otimes 1$
    \[
        J^{a_1}_{n_1} \dotsb J^{a_m}_{n_m} v_{k}
    \]
    where $n_1\leq \dotsb \leq n_m<0$ and if $n_i = n_{i+1}$ then $a_i\leq a_{i+1}$
\end{frame}

\begin{frame}{Kac-Moody Factorization Algebra}

    \begin{block}{Theorem (CG)}
        $U(1)$-invariant holomorphic prefactorization algebras on $\C$ with additional assumptions produce vertex algebras.
    \end{block}


    Define the precosheaf of dg-Lie algebras on a Riemann surface $X$
    \[
        \mathcal{G}:= \Omega^{0,*}_c \otimes \mathfrak{g}.
    \]
    For $h$ a symmetric bilinear form on $\mathfrak{g}$ define the central extension
    \[
        0 \to \C K \to \hat{\mathcal{G}} \to \mathcal{G} \to 0
    \]
    defined on $U$ by
    \[
        [\alpha \otimes X,\beta\otimes Y] = \alpha \wedge \beta \otimes [X,Y] - h(X,Y) \int_{U} \alpha \partial \beta  K
    \]

\end{frame}

\begin{frame}{Coinvariants}
For $V_{k}(\mathfrak{g})$ there is simple description of coinvariants (Frenkel--Ben-Zvi). Let $(M_i)$ be a collection of $V_{k}(\mathfrak{g})$ modules inserted at $\mathbf{x}=(x_1,\dots,x_n)$.
\[
\rho: \mathfrak{g}_{out}(X,\mathbf{x})= \mathfrak{g} \otimes \Gamma(X\setminus \mathbf{x}) \to \mathfrak{g}((z_1)) \oplus \cdots \oplus \mathfrak{g}((z_n))
\]which gives a map 
\[
\rho: \mathfrak{g}_{out} \to U(\hat{\mathfrak{g}}_k) \otimes \cdots \otimes U(\hat{\mathfrak{g}}_k).
\]
The space of coinvariants can be defined as 
\[
H_{V_{k}(\mathfrak{g})}(X,(x_i),(M_i))_{i=1}^n:=H_0(\mathfrak{g}_{out},M_1 \otimes \cdots \otimes M_n)
\]

\begin{frame}{Derived Weyl Modules}
    For $X$ a Riemann surface and $M\in \mathfrak{g}\mbox{-mod}$ we define
    \[
        \mathcal{M} =\Omega^{0,*}\otimes \mathcal{E}^\bullet \otimes M
    \]
    where $\mathcal{E}^\bullet$ is a resolution of the skyscraper sheaf at $x \in X$. $\mathcal{M}$ is a dg-mod for $\hat{\mathcal{G}}$, for $\alpha \otimes X \in \hat{\mathcal{G}}(U)$, $\beta \otimes e \otimes m \in \mathcal{M}(U)$
    \[
    \alpha \otimes X \cdot \beta \otimes e \otimes m = \alpha \wedge \beta \otimes e \otimes X \cdot m
    \]
    \pause
     We can define the following precosheaf of dg-Lie Algebras
    \[
        (\hat{\mathcal{G}}   \ltimes \mathcal{M})_c
    \]
    Taking the factorization envelope we have a prefactorization algebra
    \[
        \mathcal{F}_M = \mathrm{Sym}((\hat{\mathcal{G}}   \ltimes \mathcal{M})_c[1],d_{CE}+\bar{\partial}+d_{M}).
    \]
    We have a $\Z_{\geq 0}$ grading of $\mathcal{F}_M$ by the number of products of elements in $M$. We will denote the degree 1 piece $\mathcal{F}_M^1$.
\end{frame}

\begin{frame}{Derived Weyl Modules}
    
    \begin{block}{Proposition (Gwilliam--Oren--Szczesny--Williams)}
        For $D$ a disc at $x \in X$
        \[
            H^*(\mathcal{F}_M^{1}(D))\cong W(M)
        \]
    \end{block}
    Since we are taking the degree one piece we see that
    \[
        H^*(\mathcal{F}_M^1(D))= H_*^{Lie}(\hat{\mathcal{G}}(D) , \mathcal{M}(D))
    \]
    where 
    \[H^*(\mathcal{M}(D)) \cong M \quad H^*(\hat{\mathcal{G}}(D))\cong \mathfrak{g} \otimes \Omega_{hol}^1(D)^{\vee}\cong z^{-1}\mathfrak{g}[z^{-1}].\] 
    Using the spectral sequence for the filtration by symmetric degree \[
    H^*(\mathcal{F}^1_{M}(D)) \cong \mbox{Sym}(z^{-1}\mathfrak{g}[z^{-1}]) \otimes M \cong W(M).
    \].
\end{frame}





\begin{frame}{Coinvariants}

    If we construct $\mathcal{F}_M$ for $M$ inserted at $x$ on a compact Riemann surface $X$ then
    \begin{align*}
    H^0(\mathcal{F}^{1}(X)) \cong & H_0(\Omega^{0,*}(X) \otimes \mathfrak{g} \oplus \C K ,\mathcal{M}(X) ) \\
     \cong & H_0(R\Gamma(X,\mathcal{O}_X)\otimes \mathfrak{g} \oplus \C K,\mathcal{M}(X) ).
    \end{align*}
    
    \begin{block}{Theorem (Gwilliam--Oren--Szczesny--Williams)}
        On a compact Riemann surface $X$

        \[
         H_0 ( \mathfrak{g} \otimes R\Gamma(X, \mathcal{O}_X), \mathcal{M}(X) ) \simeq H_0(\mathfrak{g}_{out}, M)
          \]
    \end{block}


\end{frame}

\begin{frame}{Coinvariants via Factorization Homology}

For one module inserted at $x\in X$ we have the formal Cech cover $D_x,X\setminus\{x\}$. $\mathcal{F}_M^1$ computes the following Lie algebra homology 
\[
H_*(\hat{\mathcal{G}}(X),\mathcal{M}(X))
\]
where $H^*(\mathcal{M}(X))\cong M$. By Hocschild-Serre with $\mathfrak{h}=\hat{\mathcal{G}}(D)$ we can instead compute 
\[
H_*(\hat{\mathcal{G}}(X)/\hat{\mathcal{G}}(D),H_*(\hat{\mathcal{G}}(D),\mathcal{M}(X))).
\]
Schematically 
\[
H^*(\hat{\mathcal{G}}(X)/\hat{\mathcal{G}}(D) )\cong \mathfrak{g}_{out}, \quad H_*(\hat{\mathcal{G}}(D),\mathcal{M}(X))\cong W(M).
\]
\end{frame}

\begin{frame}{Raviolo Vertex Algebra}
\begin{itemize}
    \item Introduced by Garner--Williams to model the algebraic structure of topological-holomorphic field theories in 3-dimensions.
    \item Vertex algebras had a multiplication parameterized by the formal punctured disc $\mathbb{D}^\times$ which is affine, $\mathrm{Spec}(\mathbb{D}^\times)= \C((z))$. 
    \item Raviolo vertex algebras have an extra topological direction, allowing operators to move past each other in the complex direction. 
    \item The formal raviolo $\mathbb{R}\mathrm{av}=\mathbb{D}\cup_{\mathbb{D}^\times} \mathbb{D}$ is not affine. Our "poles" are $\Omega^n_z$ are in degree 1.

\begin{frame}{Raviolo Vertex Algebra}

\begin{frame}{Topological-Holomorphic (TH) Prefactorization Algebras}
    \begin{block}{Topological-Holomorphic Translation-Invariant Prefactorization Algebra on $\C \times \R$}
         A translation-invariant prefactorization algebra on $\C \times \R$ is topological-holomoprhic translation-invariant if it has derivations $\eta,\xi$ of cohomological degree $-1$ such that
                  \[
                      d\eta= \frac{\partial}{\partial \bar{z}}  \qquad d\xi = \frac{\partial}{\partial t}
                  \]
                  \[
                      [\eta,\xi]=0
                  \]
                  \[
                      [\eta,\frac{\partial}{\partial \bar{z}}] = 0 \qquad [\xi,\frac{\partial}{\partial t}]=0
                  \]
        where $d$ is the natural differential on the dg-Lie algebra $\mbox{Der}(\mathcal{F})$.
    \end{block}
\end{frame}

\begin{frame}{Dolbeaut-de Rham Cohomology}
    \begin{itemize}

        \item The Dolbeaut-de Rham complex on $\C^n \times \R^m$ is denoted $\mathcal{A}^{0;*}(\C^n\times \R^m)$.
        \item $\mathcal{A}^{0;n}(\C^n\times \R^m)$ is the subspace of $n$-forms generated by $dt_i$ and $d\bar{z}_j$.

\end{frame}

\begin{frame}{Locality}
    Since we are integrating an algebra over the cooperad $\mathcal{A}^{0;*}(B^3)$, when $|z-w|<|w|,|z|$ we can integrate $m_{w,0}(m_{z,w}(a,b),c)$ instead of $m_{z,w,0}(a,b,c)$.
    It can be shown that
    \[
        m_{\ubar{w},0}(m_{\ubar{w},\ubar{z}}(b,a),c) = c(z,w) + \sum_{i=0}^N \Omega_{z-w}^i C_i(w), \quad C_i(w) \in V\langle \langle w \rangle \rangle
    \]
    Then by direct computation one can check that
    \[
        [\mathcal{L}'_z,\mathcal{L}'_w]\Omega^{n}_{z-w} C_n(w) = \Delta^{(n)}(z-w) C_n(w)
    \]
    and
    \[
        [\mathcal{L}'_z,\mathcal{L}'_w] c(z,w) = 0.
    \]
    This shows that the fields constructed from the prefactorization algebra are mutually local.
\end{frame}

\begin{frame}{Raviolo Current Factorization Algebra}
    For $\mathfrak{g}$ a Lie algebra and $h: \mathfrak{g}^{\otimes 2} \to \C$ a $\mathfrak{g}$-invariant symmetric pairing we can define the precosheaf of dg-Lie algebras on $\C \times \R$
    \[
        \mathcal{G} :U \to (\mathcal{A}^{0;*}_c(U)\otimes \mathfrak{g}, d').
